Code Injection — Google Fonts & Head Tags
Loads typography across all pages
Settings → Advanced → Code Injection → Header

Paste this into the Header Code Injection field. It loads Space Grotesk and Inter fonts used throughout the site.

  1. Go to Settings → Advanced → Code Injection
  2. Paste the code below into the Header field
  3. Click Save
HTML — Header Injection 
<!-- AlphaCreative Fonts -->
<link rel="preconnect" href="https://fonts.googleapis.com" />
<link rel="preconnect" href="https://fonts.gstatic.com" crossorigin />
<link href="https://fonts.googleapis.com/css2?family=Space+Grotesk:wght@400;500;600;700&family=Inter:wght@300;400;500;600;700&display=swap" rel="stylesheet" />
<!-- Prevent FOUT -->
<style>body { opacity: 0; transition: opacity 0.3s; }</style>
<script>document.addEventListener('DOMContentLoaded', function(){ document.body.style.opacity = '1'; });</script>

Pages → + New Page
  1. Go to Pages in the left sidebar → click + New Page
  2. Choose Blank layout (no pre-built sections)
  3. Name it "Home" and set it as your homepage (Pages → gear icon → Set as Homepage)

Disabling the built-in header/footer ensures no style conflicts. Your custom navbar and footer from the code blocks will render instead.

JavaScript — Animations Engine (add once, at end of page)
🎬
Add this as the LAST Code Block on the page. It powers scroll progress, parallax, particles, reveal animations, counter animations, and card tilt.
JavaScript — Animations Engine 
<script>
(function() {
  'use strict';

  /* ── Scroll Progress Bar ── */
  const progressBar = document.getElementById('scroll-progress');
  function updateProgress() {
    const scrollTop = window.scrollY;
    const docH = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (docH > 0 ? (scrollTop / docH * 100) : 0) + '%';
  }

  /* ── Navbar scroll class ── */
  const navbar = document.querySelector('.navbar');
  function updateNavbar() {
    navbar.classList.toggle('scrolled', window.scrollY > 60);
  }

  /* ── Parallax on Hero ── */
  const heroSection = document.querySelector('.hero');
  function updateParallax() {
    if (!heroSection) return;
    const sy = window.scrollY;
    heroSection.style.backgroundPositionY = (sy * 0.25) + 'px';
    const heroContainer = heroSection.querySelector('.container');
    if (heroContainer) {
      heroContainer.style.transform = `translateY(${sy * 0.08}px)`;
    }
  }

  window.addEventListener('scroll', () => {
    updateProgress();
    updateNavbar();
    updateParallax();
  }, { passive: true });

  /* ── Canvas Particle System ── */
  const canvas = document.getElementById('hero-canvas');
  if (canvas) {
    const ctx = canvas.getContext('2d');
    let W, H, particles = [], animId;

    function resize() {
      W = canvas.width = canvas.offsetWidth;
      H = canvas.height = canvas.offsetHeight;
    }

    function Particle() {
      this.reset = function() {
        this.x = Math.random() * W;
        this.y = Math.random() * H;
        this.r = Math.random() * 1.4 + 0.3;
        this.vx = (Math.random() - 0.5) * 0.18;
        this.vy = (Math.random() - 0.5) * 0.18;
        this.alpha = Math.random() * 0.45 + 0.08;
        this.life = 0;
        this.maxLife = Math.random() * 300 + 200;
      };
      this.reset();
      this.life = Math.random() * this.maxLife; // stagger start
    }

    function initParticles() {
      const count = Math.floor((W * H) / 7000);
      particles = [];
      for (let i = 0; i < count; i++) {
        const p = new Particle();
        particles.push(p);
      }
    }

    function drawFrame() {
      ctx.clearRect(0, 0, W, H);
      particles.forEach(p => {
        p.x += p.vx; p.y += p.vy; p.life++;
        // fade in/out
        const progress = p.life / p.maxLife;
        const fade = progress < 0.15 ? progress / 0.15 :
                     progress > 0.85 ? (1 - progress) / 0.15 : 1;
        if (p.life >= p.maxLife) p.reset();
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(245,184,0,${p.alpha * fade})`;
        ctx.fill();
      });
      // Draw subtle connecting lines between close particles
      for (let i = 0; i < particles.length; i++) {
        for (let j = i + 1; j < particles.length; j++) {
          const dx = particles[i].x - particles[j].x;
          const dy = particles[i].y - particles[j].y;
          const dist = Math.sqrt(dx*dx + dy*dy);
          if (dist < 90) {
            const progress_i = particles[i].life / particles[i].maxLife;
            const fade_i = progress_i < 0.15 ? progress_i/0.15 : progress_i > 0.85 ? (1-progress_i)/0.15 : 1;
            ctx.beginPath();
            ctx.moveTo(particles[i].x, particles[i].y);
            ctx.lineTo(particles[j].x, particles[j].y);
            ctx.strokeStyle = `rgba(245,184,0,${0.06 * (1 - dist/90) * fade_i})`;
            ctx.lineWidth = 0.5;
            ctx.stroke();
          }
        }
      }
      animId = requestAnimationFrame(drawFrame);
    }

    window.addEventListener('resize', () => { resize(); initParticles(); }, { passive: true });
    resize();
    initParticles();
    drawFrame();
  }

  /* ── Intersection Observer: Reveal ── */
  const revealEls = document.querySelectorAll('[data-reveal]');
  if (revealEls.length) {
    const revealObs = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          entry.target.classList.add('revealed');
          revealObs.unobserve(entry.target);
        }
      });
    }, { threshold: 0.12, rootMargin: '0px 0px -40px 0px' });
    revealEls.forEach(el => revealObs.observe(el));
  }

  /* ── Counter Animation ── */
  function animateCount(el, target, duration, prefix, suffix) {
    prefix = prefix || '';
    suffix = suffix || '';
    const start = performance.now();
    const isInt = Number.isInteger(target);
    function tick(now) {
      const elapsed = now - start;
      const progress = Math.min(elapsed / duration, 1);
      // Ease out expo
      const eased = progress === 1 ? 1 : 1 - Math.pow(2, -10 * progress);
      const current = target * eased;
      el.textContent = prefix + (isInt ? Math.round(current) : current.toFixed(1)) + suffix;
      if (progress < 1) requestAnimationFrame(tick);
    }
    requestAnimationFrame(tick);
  }

  const countEls = document.querySelectorAll('[data-count]');
  if (countEls.length) {
    const countObs = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const el = entry.target;
          const target = parseFloat(el.dataset.count);
          const prefix = el.dataset.prefix || '';
          const suffix = el.dataset.suffix || '';
          animateCount(el, target, 1400, prefix, suffix);
          countObs.unobserve(el);
        }
      });
    }, { threshold: 0.5 });
    countEls.forEach(el => countObs.observe(el));
  }

  /* ── Package Card 3D Tilt ── */
  document.querySelectorAll('.pkg-card').forEach(card => {
    card.addEventListener('mousemove', e => {
      const rect = card.getBoundingClientRect();
      const cx = rect.left + rect.width / 2;
      const cy = rect.top + rect.height / 2;
      const dx = (e.clientX - cx) / (rect.width / 2);
      const dy = (e.clientY - cy) / (rect.height / 2);
      card.style.transform = `perspective(800px) rotateY(${dx * 4}deg) rotateX(${-dy * 3}deg) translateZ(8px)`;
    });
    card.addEventListener('mouseleave', () => {
      card.style.transform = '';
    });
  });

  /* Init */
  updateProgress();
  updateNavbar();

})();
</script>
